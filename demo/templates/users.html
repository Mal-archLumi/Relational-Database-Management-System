-- Table creation with encrypted column
CREATE TABLE users (
    id INT PRIMARY KEY,
    username VARCHAR(50) UNIQUE,
    email VARCHAR(100),
    password TEXT ENCRYPTED,  -- ‚≠ê This column is encrypted!
    age INT,
    is_active BOOLEAN
)

-- Insert (automatic encryption)
INSERT INTO users VALUES (1, 'alice', 'alice@example.com', 'secret123', 25, true)

-- Select (automatic decryption)
SELECT id, username, email, age, is_active FROM users
